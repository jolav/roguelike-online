## CONTROLS 

(&uarr;) or (w) = Up  
(&darr;) or (s) = Down  
(&larr;) or (a) = Left  
(&rarr;) or (d) = Right  
(q) = Loot  
(t) = Skip Turn  
(1) = Eat Food  
(3) = Heal with medical  
(r) = Select Next Foe  
(f) = Shoot firearm
